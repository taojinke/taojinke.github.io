Sequelize.js 提供对MySQL，MariaDB，SQLite 和PostgreSQL 数据库的简单访问，通过映射数据库条目到对象，或者对象到数据库条目。简而言之，就是 ORM（Object-Relational-Mapper）。Sequelize.js 完全是使用 JavaScript 编写，适用于 Node.js 的环境。

简单使用

创建 Sequelize 实例：

var sequelize = new Sequelize(&apos;database&apos;, &apos;username&apos;[, &apos;password&apos;])

指定 host/port：

var sequelize = new Sequelize(&apos;database&apos;, &apos;username&apos;, &apos;password&apos;, {
  host: &quot;my.server.tld&quot;,  port: 12345
})

没有密码的情况：

var sequelize = new Sequelize(&apos;database&apos;, &apos;username&apos;)
// or
var sequelize = new Sequelize(&apos;database&apos;, &apos;username&apos;, null)

使用连接字符串：

var sequelize = new Sequelize(&apos;mysql://user:pass@example.com:9821/dbname&apos;, {  
    // Look to the next section for possible options
})
