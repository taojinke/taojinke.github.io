//- &#x5B89;&#x88C5;&#x63D2;&#x4EF6;&#x5230;&#x9879;&#x76EE;&#x76EE;&#x5F55;&#x5185;

完成上面两步后，会在我们的项目内生成一个 package.json 文件以及一个 node_modules 目录

创建配置文件 gulpfile.js

在项目根目录内创建一个 gulpfile.js 文���(必须是这个文件名哟~)，内容就是上面几个插件的配置信息： 

var gulp = require(&apos;gulp&apos;);
var concat = require(&apos;gulp-concat&apos;);                            //- &#x591A;&#x4E2A;&#x6587;&#x4EF6;&#x5408;&#x5E76;&#x4E3A;&#x4E00;&#x4E2A;&#xFF1B;
var minifyCss = require(&apos;gulp-minify-css&apos;);                     //- &#x538B;&#x7F29;CSS&#x4E3A;&#x4E00;&#x884C;&#xFF1B;
var rev = require(&apos;gulp-rev&apos;);                                  //- &#x5BF9;&#x6587;&#x4EF6;&#x540D;&#x52A0;MD5&#x540E;&#x7F00;
var revCollector = require(&apos;gulp-rev-collector&apos;);               //- &#x8DEF;&#x5F84;&#x66FF;&#x6362;
gulp.task(&apos;concat&apos;, function() {                                //- &#x521B;&#x5EFA;&#x4E00;&#x4E2A;&#x540D;&#x4E3A; concat &#x7684; task
    gulp.src([&apos;./css/wap_v3.1.css&apos;, &apos;./css/wap_v3.1.3.css&apos;])    //- &#x9700;&#x8981;&#x5904;&#x7406;&#x7684;css&#x6587;&#x4EF6;&#xFF0C;&#x653E;&#x5230;&#x4E00;&#x4E2A;&#x5B57;&#x7B26;&#x4E32;&#x6570;&#x7EC4;&#x91CC;
        .pipe(concat(&apos;wap.min.css&apos;))                            //- &#x5408;&#x5E76;&#x540E;&#x7684;&#x6587;&#x4EF6;&#x540D;
        .pipe(minifyCss())                                      //- &#x538B;&#x7F29;&#x5904;&#x7406;&#x6210;&#x4E00;&#x884C;
        .pipe(rev())                                            //- &#x6587;&#x4EF6;&#x540D;&#x52A0;MD5&#x540E;&#x7F00;
        .pipe(gulp.dest(&apos;./css&apos;))                               //- &#x8F93;&#x51FA;&#x6587;&#x4EF6;&#x672C;&#x5730;
        .pipe(rev.manifest())                                   //- &#x751F;&#x6210;&#x4E00;&#x4E2A;rev-manifest.json
        .pipe(gulp.dest(&apos;./rev&apos;));                              //- &#x5C06; rev-manifest.json &#x4FDD;&#x5B58;&#x5230; rev &#x76EE;&#x5F55;&#x5185;
});
gulp.task(&apos;rev&apos;, function() {
    gulp.src([&apos;./rev/*.json&apos;, &apos;./application/**/header.php&apos;])   //- &#x8BFB;&#x53D6; rev-manifest.json &#x6587;&#x4EF6;&#x4EE5;&#x53CA;&#x9700;&#x8981;&#x8FDB;&#x884C;css&#x540D;&#x66FF;&#x6362;&#x7684;&#x6587;&#x4EF6;
        .pipe(revCollector())                                   //- &#x6267;&#x884C;&#x6587;&#x4EF6;&#x5185;css&#x540D;&#x7684;&#x66FF;&#x6362;
        .pipe(gulp.dest(&apos;./application/&apos;));                     //- &#x66FF;&#x6362;&#x540E;&#x7684;&#x6587;&#x4EF6;&#x8F93;&#x51FA;&#x7684;&#x76EE;&#x5F55;
});
gulp.task(&apos;default&apos;, [&apos;concat&apos;, &apos;rev&apos;]);

运行结果

通过以上的配置之后，就可以开始运行 gulp 对我们的项目进行相关的操作啦；

使用 gulp 命令，运行 Gulp.js 构建程序

首先运行 concat 这个 task 生成一个 rev-manifest.json 文件

然后再运行 rev 这个 task 替换掉文件中引入的 css

rev-manifest.json文件内容： 

{
  &quot;wap.min.css&quot;: &quot;wap.min-c49f62a273.css&quot;
}

header.php替换前后对比：

替换前： <link rel="stylesheet" href="/css/wap.min.css" />

替换后： <link rel="stylesheet" href="/css/wap.min-c49f62a273.css" />

gulp-rev-collector 进行文件路径替换是依据 rev-manifest.json 的，所以要先成生 .json 文件，然后再进行替换；
