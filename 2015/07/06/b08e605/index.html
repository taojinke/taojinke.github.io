centOS下编译安装memcached -

By edwin

发表于 2015-07-06

文章目录

介绍

memcached依赖于libevent库，所以需要先安装libevent

安装libevent

libevent官网

//&#x4E0B;&#x8F7D;
wget http://sourceforge.net/projects/levent/files/libevent/libevent-2.0/libevent-2.0.22-stable.tar.gz
//&#x89E3;&#x538B;
tar -zxvf libevent-2.0.22-stable.tar.gz
//&#x914D;&#x7F6E;
cd libevent-2.0.22-stable
./configure --prefix=/usr/local/libevent/
//&#x7F16;&#x8BD1;&#x5B89;&#x88C5;
make
make install

安装memcached

memcached官网

//&#x4E0B;&#x8F7D;&#x6700;&#x65B0;&#x7684;stable&#x7248;&#x672C;
wget http://memcached.org/latest
//&#x89E3;&#x538B;
tar -zxvf memcached-1.4.24.tar.gz
cd memcached-1.4.24
//&#x914D;&#x7F6E;&#xFF0C;&#x5047;&#x5982;liv&#x5B89;&#x88C5;&#x5728;&#x5176;&#x4ED6;&#x76EE;&#x5F55;&#xFF0C;&#x81EA;&#x884C;&#x4FEE;&#x6539;
./configure --prefix=/usr/local/memcached/ --with-libevent=/usr/local/libevent/
//&#x7F16;&#x8BD1;&#x5B89;&#x88C5;
make&amp;make install

启动memcached

//&#x5185;&#x5B58;&#x9ED8;&#x8BA4;64M&#xFF0C;&#x6700;&#x5927;2G&#xFF0C;&#x73B0;&#x5728;&#x914D;&#x7F6E;16M
/usr/local/memcached/bin/memcached -d -m 16 -l 127.0.0.1 -p 11211 -u root -P /tmp/memcached.pid

无提示则启动成功。

防火墙开启11211端口号：

vim /etc/sysconfig/iptables
//&#x52A0;&#x5165;&#x4EE5;&#x4E0B;&#x4EE3;&#x7801;
-A INPUT -m state --state NEW -m tcp -p tcp --dport 11211 -j ACCEPT
//&#x91CD;&#x542F;&#x9632;&#x706B;&#x5899;
service iptables restart

安装telnet

//&#x68C0;&#x67E5;&#x662F;&#x5426;&#x5DF2;&#x7ECF;&#x5B89;&#x88C5;telnet
rpm -qa |grep telnet
//telnet&#x4F9D;&#x8D56;&#x4E8E;xinetd
yum install xinetd
yum install telnet -server
yum install telnet
//&#x914D;&#x7F6E;telnet&#xFF0C;&#x8F93;&#x5165;&#x4EE5;&#x4E0B;&#x547D;&#x4EE4;&#x540E;&#x73B0;&#x5728;telnet&#x9009;&#x62E9;OK&#x5373;&#x53EF;

//&#x628A;service telnet&#x7684;disable&#x6539;&#x4E3A;no
vi /etc/xinetd.d/telnet
//xinetd&#x91CD;&#x542F;
service xinetd restart
//&#x9632;&#x706B;&#x5899;&#x6253;&#x5F00;23&#x7AEF;&#x53E3;
iptables -I INPUT -p tcp --dport 23 -jACCEPT

service iptables save
service iptables restart
