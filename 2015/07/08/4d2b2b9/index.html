mongodb备份脚本：

#!/bin/sh

TIME=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR=/data/backup/mongo/local
COMMAND=/data/server/mongodb/bin/mongodump
TTL=10
LOCAL_NAME=$(/bin/hostname)

#zip
BACKUP(){
    if [ ! -d ${BACKUP_DIR}/${LOCAL_NAME}_mongo_${TIME} ];then
        mkdir -p ${BACKUP_DIR}/${LOCAL_NAME}_mongo_${TIME}
    fi

    ${COMMAND} -h 10.252.128.164 -p 27017 -o ${BACKUP_DIR}/${LOCAL_NAME}_mongo_${TIME} &amp;gt;/var/log/mongodump_out.log 2&amp;gt;&amp;amp;1
    cd ${BACKUP_DIR}
    tar zcf ${LOCAL_NAME}_mongo_${TIME}.tar.gz ${LOCAL_NAME}_mongo_${TIME}
    if [ &quot;$?&quot; -eq 0 ];then
        rm -rf ${LOCAL_NAME}_mongo_${TIME}
    fi
}

#clear old backup
DELETE(){
    find ${BACKUP_DIR} -type f -name &quot;*.tar.gz&quot; -ctime +$TTL -exec rm -rf {} ;
}


if [ ! -d ${BACKUP_DIR} ]; then
    mkdir -p ${BACKUP_DIR}
fi

if [ -f ${COMMAND} ];then
    BACKUP
    DELETE
fi

redis备份脚本：

#!/bin/sh
#
#defined redis1 data path like: /data/server/redis/data/dump_6379.rdb ----&amp;gt; port:6379
#defined redis2 data path like: /data/server/redis/data/dump_6380.rdb ----&amp;gt; port:6380

if [[ $UID -ne 0 ]]; then
        echo &quot;You must run as root&quot;
        exit
fi

LOCALTIME=`date +%Y%m%d_%H%M`
HOSTNAME=$(/bin/hostname)
PORT=($(netstat -lntp | grep redis-server | awk &apos;{print $4}&apos; | awk &apos;{split($0,ports,&quot;:&quot;);print ports[length(ports)]}&apos; | sort | uniq))
BACKUP_PATH=/data/backup/redis/local

if [[ -z `pgrep redis-server` ]];then
    echo -e &quot;n33[40;31m No redis server is running33[0mn&quot;
    exit
fi

#zip
for i in ${PORT[@]};do
    cd /data/server/redis/data
    tar czvf ${HOSTNAME}_redis_${i}_${LOCALTIME}.tar.gz dump_${i}.rdb
done

if [ ! -d ${BACKUP_PATH} ]; then
    mkdir -p ${BACKUP_PATH}
fi

#move
for i in ${PORT[@]};do
    mv /data/server/redis/data/${HOSTNAME}_redis_${i}_${LOCALTIME}.tar.gz ${BACKUP_PATH}
done

# clear old backup
find ${BACKUP_PATH}/ -type f -name &quot;*.tar.gz&quot;  -ctime +10 -exec rm -rf {} ;
