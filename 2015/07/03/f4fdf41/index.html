Android C++ 引用计数介绍, part 1 -

By edwin

发表于 2015-07-03

文章目录

任何一个基于Android 的native框架做开发的人都不可避免的会碰到一些几乎到处使用的native层C++工具类。 sp (或者称为 StrongPointer )，就是其中之一。了解它是如何工作非常重要，这样你才能更清晰的理解代码，并且写出精简、没有资源泄漏的代码。在这篇文章中，我们就要根据实例来了解 sp 的基本概念和使用方法。 

#include &lt;utils/RefBase.h&gt;
#include &lt;utils/Log.h&gt;
#include &lt;cstdlib&gt;
#include &lt;cassert&gt;

using namespace android;

// &#x6211;&#x4EEC;&#x81EA;&#x5B9A;&#x4E49;&#x4E86;&#x4E00;&#x4E2A; Memory &#x7C7B;&#x7EE7;&#x627F;&#x81EA; RefBase [1]&#xFF0C;&#x4EE5;&#x4FDD;&#x8BC1;&#x5176;&#x62E5;&#x6709;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x7684;&#x80FD;&#x529B;&#xFF0C;&#x5E76;&#x4E14;&#x80FD;&#x88AB; sp&lt;T&gt; [2] &#x6A21;&#x677F;&#x7C7B;&#x63A5;&#x6536;&#xFF0C;&#x8FD9;&#x91CC; sp &#x6307;&#x7684;&#x662F;&#x5F3A;&#x5F15;&#x7528;&#x6307;&#x9488;&#x3002;
// [1]https://android.googlesource.com/platform/frameworks/native/+/jb-mr1-dev/include/utils/RefBase.h
// [2]https://android.googlesource.com/platform/frameworks/native/+/jb-mr1-dev/include/utils/StrongPointer.h

class Memory: public RefBase {
public:
    Memory(int size) : mSize(size), mData(NULL) {
        ALOGD(&quot;        Memory constructor %p &quot;,this);
    }
    virtual ~Memory() {
        ALOGD(&quot;        Memory destructor %p&quot;, this);
        if (mData)  free(mData);
    }
    virtual void onFirstRef() {
        ALOGD(&quot;        onFirstRef on %p&quot;,this);
        mData = malloc(mSize);
    }
    int size() {return mSize;}
private:
    int mSize;
    void *mData;
};

// &#x7528;&#x4E8E;&#x8F93;&#x51FA;&#x6807;&#x8BB0;log
#define L(N)   ALOGD(&quot;LINE %d TRIGGER:&quot;,N);
// &#x8F93;&#x51FA;object&#x7684;&#x5F3A;&#x5F15;&#x7528;&#x6570;&#x91CF;
#define C(obj) ALOGD(&quot;        Count of %p : %d&quot;, (void*)obj, obj-&gt;getStrongCount());

int main()
{
    {
        // &#x521B;&#x5EFA;&#x4E00;&#x4E2A; Memory &#x7684;&#x5B9E;&#x4F8B;&#xFF0C;&#x5E76;&#x4E14;&#x8D4B;&#x503C;&#x7ED9;&#x4E00;&#x4E2A;&#x539F;&#x59CB;&#x6307;&#x9488;&#x3002;
        L(1)
        Memory *m1 = new Memory(4);
        // &#x4F7F;&#x7528; sp(T* other) &#x6784;&#x9020;&#x51FD;&#x6570;&#x521B;&#x5EFA;&#x4E00;&#x4E2A;&#x5F3A;&#x5F15;&#x7528;&#x6307;&#x9488;&#xFF0C;&#x8FD9;&#x6837;&#x4F1A;&#x4F7F; m1 &#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x503C;&#x52A0;1&#xFF0C;&#x5E76;&#x4E14;&#x8C03;&#x7528; m1::onFirstRef&#xFF0C;&#x8FD9;&#x91CC;&#x4F60;&#x53EF;&#x4EE5;&#x505A;&#x4E00;&#x4E9B;&#x5EF6;&#x8FDF;&#x521D;&#x59CB;&#x5316;&#x64CD;&#x4F5C;&#x3002;
        L(2)
        sp&lt;Memory&gt; spm1 = m1;
        C(m1);
        // &#x901A;&#x5E38;&#xFF0C;&#x6211;&#x4EEC;&#x4F1A;&#x628A;&#x4E0A;&#x9762;&#x4E24;&#x6B65;&#x5408;&#x5E76;&#x4E3A;&#x4E00;&#x884C;&#x4EE3;&#x7801;&#x3002;
        // &#x7136;&#x540E;&#x6211;&#x4EEC;&#x521B;&#x5EFA;&#x53E6;&#x5916;&#x4E00;&#x4E2A;&#x5F3A;&#x5F15;&#x7528;&#x6307;&#x9488;&#xFF0C;spm2, &#x5E76;&#x4E14;&#x521D;&#x59CB;&#x5316;&#x3002;
        // &#x8981;&#x62FF;&#x5230;&#x539F;&#x59CB;&#x7684;object&#xFF0C;&#x53EF;&#x4EE5;&#x4F7F;&#x7528; sp&lt;T&gt;::get() &#x65B9;&#x6CD5;&#x3002;
        L(3)
        sp&lt;Memory&gt; spm2 = new Memory(128);
        Memory *m2 = spm2.get();
        // &#x8981;&#x60F3;&#x8C03;&#x7528;&#x539F;object&#x4E2D;&#x7684;&#x65B9;&#x6CD5;&#xFF0C;&#x4F7F;&#x7528; sp &#x5C31;&#x50CF;&#x7528;&#x539F;&#x59CB;&#x6307;&#x9488;&#x4E00;&#x6837;&#x65B9;&#x4FBF;&#x3002;
        int size = spm2-&gt;size();
        // &#x521B;&#x5EFA;&#x7B2C;&#x4E09;&#x4E2A; sp, spm3, &#x8FD9;&#x91CC;&#x4F1A;&#x8C03;&#x7528;&#x6784;&#x9020;&#x51FD;&#x6570; sp(const sp&lt;T&gt;&amp; other)&#xFF0C;&#x8FD9;&#x6837;&#x4F1A;&#x4F7F; spm1 &#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x52A0;1&#xFF0C;&#x73B0;&#x5728; m1 &#x88AB;&#x4E24;&#x4E2A;&#x5F3A;&#x5F15;&#x7528;&#x6307;&#x9488;&#x6307;&#x5411;&#xFF0C;spm1 &#x548C; spm3&#x3002;
        L(4)
        sp&lt;Memory&gt; spm3 = spm1;
        C(m1);
        // &#x4E0B;&#x9762;&#x8FD9;&#x6BB5;&#x4EE3;&#x7801;&#x548C; L(4) &#x5DEE;&#x4E0D;&#x591A;&#xFF0C;&#x533A;&#x522B;&#x5728;&#x4E8E; spm4 &#x7684;&#x4F5C;&#x7528;&#x57DF;&#x88AB;&#x9650;&#x5236;&#x5728;&#x4E00;&#x4E2A;&#x4EE3;&#x7801;&#x5757;&#x4E2D;&#x3002;
        L(5)
        {
            sp&lt;Memory&gt; spm4 = spm1;
            C(m1);
            // &#x8FD9;&#x91CC; m1 &#x88AB; spm1, spm3 &#x548C; spm4 &#x6307;&#x5411;&#x3002;
        }
        // &#x5230;&#x4E86;&#x8FD9;&#x91CC;&#xFF0C;spm4 &#x79BB;&#x5F00;&#x4E86;&#x4F5C;&#x7528;&#x57DF;&#x88AB;&#x9500;&#x6BC1;&#xFF0C;&#x4E0D;&#x518D;&#x6307;&#x5411; m1&#xFF0C;&#x56E0;&#x6B64; m1 &#x7684;&#x5F15;&#x7528;&#x6570;&#x53D8;&#x56DE;&#x4E86;2&#xFF0C;&#x65E2;&#x88AB; spm1 &#x548C; spm3 &#x6307;&#x5411;&#x3002;
        L(6)
        C(m1);
        // trigger sp&amp; operator = (const sp&lt;T&gt;&amp; other);
        L(7)
        // &#x5728;&#x4E0B;&#x9762;&#x8FD9;&#x884C;&#x8D4B;&#x503C;&#x4E4B;&#x524D;&#xFF0C;spm2 &#x6307;&#x5411; m2&#xFF0C;smp3 &#x6307;&#x5411; m1&#x3002;
        spm3 = spm2;
        // &#x8D4B;&#x503C;&#x4E4B;&#x540E;&#xFF0C; spm3 &#x4E0D;&#x518D;&#x6307;&#x5411; m1 &#x800C;&#x662F; &#x6307;&#x5411; m2&#xFF0C;&#x56E0;&#x6B64; m1 &#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x51CF;1&#xFF0C;m2 &#x7684;&#x52A0;1&#x3002;
        C(m1);
        C(m2);
        // spm5 &#x662F; spm1 &#x7684;&#x5F15;&#x7528;&#xFF0C;&#x6CA1;&#x6709;&#x65B0;&#x7684;&#x5BF9;&#x8C61;&#x521B;&#x5EFA;&#xFF0C;&#x56E0;&#x6B64; m1 &#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x6CA1;&#x53D8;&#x3002;
        L(8)
        sp&lt;Memory&gt; &amp;spm5 = spm1;
        C(m1);
        // &#x6211;&#x4EEC;&#x4E5F;&#x53EF;&#x4EE5;&#x521B;&#x5EFA;&#x4E00;&#x4E2A;&#x667A;&#x80FD;&#x6307;&#x9488;&#x521D;&#x59CB;&#x4E3A;&#x7A7A;&#xFF0C;&#x540E;&#x7EED;&#x518D;&#x8D4B;&#x503C;&#x3002;&#x6211;&#x4EEC;&#x4E5F;&#x53EF;&#x4EE5;&#x8C03;&#x7528; sp::clear() &#x6765;&#x663E;&#x5F0F;&#x7684;&#x6E05;&#x9664;&#x5F15;&#x7528;&#x3002;
        L(9)
        sp&lt;Memory&gt; spm6;
        assert(spm6.get() == NULL);
        spm6 = spm1;
        C(m1);
        L(10)
        spm6.clear();
        assert(spm6.get() == NULL);
        C(m1);
    }
    // &#x4E0A;&#x8FF0;&#x4EE3;&#x7801;&#x5757;&#x7ED3;&#x675F;&#x4E4B;&#x540E;&#xFF0C;&#x6240;&#x6709;&#x7684;&#x667A;&#x80FD;&#x6307;&#x9488;&#x90FD;&#x8131;&#x79BB;&#x4E86;&#x4F5C;&#x7528;&#x57DF;&#xFF0C;&#x56E0;&#x6B64;&#x5B83;&#x4EEC;&#x90FD;&#x5C06;&#x4F1A;&#x88AB;&#x9500;&#x6BC1;&#xFF0C;&#x5E76;&#x4E14;&#x89E6;&#x53D1;&#x5B83;&#x4EEC;&#x5404;&#x81EA;&#x6307;&#x5411;&#x7684;object&#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x51CF;1&#x3002;&#x4F8B;&#x5982;&#xFF0C;&#x5F53; spm1 &#x548C; spm6 &#x90FD;&#x9500;&#x6BC1;&#x65F6;&#xFF0C;m1 &#x7684;&#x5F15;&#x7528;&#x8BA1;&#x6570;&#x51CF;&#x5230;&#x4E86;0&#xFF0C;&#x7136;&#x540E;&#x5C06;&#x4F1A;&#x89E6;&#x53D1; m1 &#x7684;&#x6790;&#x6784;&#x3002;
    L(-1)
    return 0;
}
