最近完成一个简单的日志管理系统，拿出来跟大家分享一下！

主要实现的功能：

1、支持动态修改配置

2、实现统一的配置管理

3、支持文件输出、habse输出、mongodb输出

基于以上三点功能，我们下面详细说明

1、支持动态修改配置

说道支持这个功能，有个同事认为没有这个必要，他的观点是log4j的配置不需要经常变动，不需要支持这样的功能；本人的观点是“配置可以进行统一管理、而且正式机跟测试机的log4j的配置肯定会有一些差异的”，因此这个功能是必须的。

下面说一下实现

通过log4j提供的PropertyConfigurator类可以非常轻松的实现这个功能

代码如下

public class Log4jConfigListener {

 private  String log4jPath;

 /**
 * @param log4jPath the log4jPath to set
 */
 public void setLog4jPath(String log4jPath) {
 this.log4jPath = log4jPath;
 }


 /**
 * &#x88C5;&#x8F7D;log4j&#x914D;&#x7F6E;&#x6587;&#x4EF6;
* 
 * @author mrh
 * @DATE 2011-5-28
 */
 public void load() {
 // String path=&quot;config/log4j.properties&quot;;
 System.out.println(&quot;log4j configfile path=&quot; + log4j.toString());
 PropertyConfigurator.configureAndWatch(log4j.toString(), 1000);// &#x95F4;&#x9694;&#x7279;&#x5B9A;&#x65F6;&#x95F4;&#xFF0C;&#x68C0;&#x6D4B;&#x6587;&#x4EF6;&#x662F;&#x5426;&#x4FEE;&#x6539;&#xFF0C;&#x81EA;&#x52A8;&#x91CD;&#x65B0;&#x8BFB;&#x53D6;&#x914D;&#x7F6E;
}
}

Spring中的配置

<bean class="com.jl.net.log4j.config.Log4jConfigListener" init-method="load">

<property name="log4jPath" value="WEB-INF/classes/log4j/" /> <!—通过zookeeper实现统一配置的dataId—>

</bean>

2、统一配置管理

统一配置管理，是通过zookeeper实现的，配置形式与log4j.properties配置一致；

配置如下

log4j.rootLogger=INFO,console,HbaseAppender

console

log4j.appender.console=org.apache.log4j.ConsoleAppender

log4j.appender.console.layout=org.apache.log4j.PatternLayout

log4j.appender.console.layout.ConversionPattern=%d{ABSOLUTE} %5p %c{1}\:%L - %m%n

hbase

log4j.appender.HbaseAppender=com.jl.net.log4j.hbase.HbaseAppender

log4j.appender.HbaseAppender.layout=org.apache.log4j.PatternLayout

log4j.appender.HbaseAppender.sysName=tuid

log4j.appender.HbaseAppender.serverIp=Y

log4j.appender.HbaseAppender.zookeeper_ip=10.1.18.100,10.1.18.103,10.1.18.102

log4j.appender.HbaseAppender.zookeeper_port=2181

log4j.appender.HbaseAppender.layout.ConversionPattern=%d{yyyy-MM-dd HH:mm:ss sss} %c.%t:%L %-5p %x  %m%n

Spring中的配置

<bean class="com.jl.net.log4j.config.Log4jConfigListener" init-method="load">

<property name="log4jPath" value="cfg/jlcloud.uid.log4j" /> <!—通过zookeeper实现统一配置的dataId—>

</bean>

3、支持文件、habse、mongodb的输出

log4j本身就支持文件输出，mongodb的输出需要借助log4mongo-Java这个项目，maven的坐标是

<dependency>

<groupId>org.log4mongo</groupId>

<artifactId>log4mongo-java</artifactId>

<version>0.7.4</version>

<optional>true</optional>

</dependency>

下面介绍一下如何实现hbase的输出，通过继承log4j的AppenderSkeleton类，同时实现Runnable接口，来实现：

代码如下

package com.yck.worm.hbase;
import java.io.IOException;
import java.net.InetAddress;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;
import java.util.Queue;
import java.util.UUID;
import java.util.concurrent.ConcurrentLinkedQueue;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.ScheduledFuture;
import java.util.concurrent.TimeUnit;
import org.apache.hadoop.conf.Configuration;
import org.apache.hadoop.hbase.HBaseConfiguration;
import org.apache.hadoop.hbase.HColumnDescriptor;
import org.apache.hadoop.hbase.HConstants;
import org.apache.hadoop.hbase.HTableDescriptor;
import org.apache.hadoop.hbase.TableName;
import org.apache.hadoop.hbase.client.HBaseAdmin;
import org.apache.hadoop.hbase.client.HConnection;
import org.apache.hadoop.hbase.client.HConnectionManager;
import org.apache.hadoop.hbase.client.HTableInterface;
import org.apache.hadoop.hbase.client.Put;
import org.apache.hadoop.hbase.util.Bytes;
import org.apache.log4j.AppenderSkeleton;
import org.apache.log4j.spi.LoggingEvent;
import com.yck.worm.logging.WLogger;
/**
 * &#x5C06;&#x65E5;&#x5FD7; &#x8F93;&#x51FA;&#x5230;HBase&#x91CC;
 * @author mrh
 *
 */
public class HbaseAppender extends AppenderSkeleton implements Runnable {
    private static final WLogger LOGGER = WLogger.getLogger(HbaseAppender.class);
    private int batchSize = -1;
    private int period = 1000;
    private String hbLogName = &quot;jl_logs&quot;;
    private String hbLogFamily = &quot;logs&quot;;
    private Queue&lt;LoggingEvent&gt; loggingEvents;
    private ScheduledExecutorService executor;
    private ScheduledFuture&lt;?&gt; task;
    private Configuration conf;
    private HTableInterface htable;
    private HBaseAdmin admin;
    private HConnection connection;
    private String zookeeper;
    private String sysName;
    /**
     * if serverIp = Y then out put server&apos;s IP to hbase 
     */
    private String serverIp;
    /**
     * log4j&#x521D;&#x59CB;&#x8BBE;&#x7F6E;&#xFF0C;&#x542F;&#x52A8;&#x65E5;&#x5FD7;&#x5904;&#x7406;&#x8BA1;&#x5212;&#x4EFB;&#x52A1;
     */
    @Override
    public void activateOptions() {
        try {
            super.activateOptions();
            // &#x521B;&#x5EFA;&#x4E00;&#x4E2A;&#x8BA1;&#x5212;&#x4EFB;&#x52A1;&#xFF0C;&#x5E76;&#x81EA;&#x5B9A;&#x4E49;&#x7EBF;&#x7A0B;&#x540D;
            executor = Executors.newSingleThreadScheduledExecutor(new NamedThreadFactory(&quot;HbaseAppender&quot;));
            // &#x65E5;&#x5FD7;&#x961F;&#x5217;
            loggingEvents = new ConcurrentLinkedQueue&lt;LoggingEvent&gt;();
            // &#x542F;&#x52A8;&#x8BA1;&#x5212;&#x4EFB;&#x52A1;&#xFF0C;&#x5982;&#x679C;run&#x51FD;&#x6570;&#x6709;&#x5F02;&#x5E38;&#x4EFB;&#x52A1;&#x5C06;&#x4E2D;&#x65AD;&#xFF01;
            task = executor.scheduleWithFixedDelay(this, period, period,TimeUnit.MILLISECONDS);
            if (&quot;Y&quot;.equalsIgnoreCase(this.serverIp)) {
                InetAddress addr = InetAddress.getLocalHost();
                this.serverIp = addr.getHostAddress().toString();//&#x83B7;&#x5F97;&#x672C;&#x673A;IP 
            }
            LOGGER.info(&quot;ActivateOptions ok!&quot;);
        } catch (Exception e) {
            LOGGER.error(&quot;Error during activateOptions: &quot; , e);
        }
    }
    /**
     * &#x521D;&#x59CB;HBASE
     *
     * @return
     */
    private boolean initHbase() {
        try {
            if (conf == null) {
                LOGGER.info(&quot;initHbase ... zookeeper address is &quot; + this.zookeeper);
                if (this.zookeeper == null || &quot;&quot;.equals(this.zookeeper)) {
                    throw new IllegalArgumentException(&quot;the zookeeper address for jlog4j&apos;s hadoop initial can not be null!&quot;);
                }
                String ip = this.zookeeper.split(&quot;:&quot;)[0];
                String port = this.zookeeper.split(&quot;:&quot;)[1];
                conf = HBaseConfiguration.create();
                conf.set(HConstants.ZOOKEEPER_QUORUM, ip);
                conf.setInt(HConstants.ZOOKEEPER_CLIENT_PORT, Integer.parseInt(port));
                this.admin  = new HBaseAdmin(conf);
                if (!this.admin.tableExists(Bytes.toBytes(hbLogName))) {
                    HTableDescriptor tableDescriptor = new HTableDescriptor(TableName.valueOf(hbLogName));
                    HColumnDescriptor columnDescriptor = new HColumnDescriptor(hbLogFamily.getBytes());
                    tableDescriptor.addFamily(columnDescriptor);
                    this.admin.createTable(tableDescriptor);
                }
                this.connection = HConnectionManager.createConnection(conf);
                this.htable = this.connection.getTable(hbLogName.getBytes());
            }
            LOGGER.error(&quot;Init Hbase success !&quot;);
            return true;
        } catch (Exception e) {
            task.cancel(false);
            executor.shutdown();
            LOGGER.error(&quot;Init Hbase fail !&quot;, e);
            return false;
        }
    }
    @Override
    public void run() {
        if (conf == null || htable == null) {
            initHbase();
        }
        try {
            if (this.batchSize &lt; 0 &amp;&amp; loggingEvents.size() &gt; 0 ) {
                this.output();
            } else 
             // &#x65E5;&#x5FD7;&#x6570;&#x636E;&#x8D85;&#x51FA;&#x6279;&#x91CF;&#x5904;&#x7406;&#x5927;&#x5C0F;
            if (loggingEvents.size() &gt; 0 &amp;&amp; loggingEvents.size() &gt; this.batchSize) {
                this.output();
            }
        } catch (Exception e) {
            LOGGER.error(&quot;HbaseAppender Error run &quot;, e);
        }
    }
    /**
     * &#x8F93;&#x51FA;&#x65E5;&#x5FD7;
     * @throws IOException
     */
    private void output() throws IOException {
        LoggingEvent event;
        List&lt;Put&gt; logs = new ArrayList&lt;Put&gt;();
        // &#x5FAA;&#x73AF;&#x5904;&#x7406;&#x65E5;&#x5FD7;&#x961F;&#x5217;
        while ((event = loggingEvents.poll()) != null) {
            try {
                // &#x5199;&#x65E5;&#x5FD7;&#x5185;&#x5BB9;
                SimpleDateFormat sdf = new SimpleDateFormat(&quot;yyyy-MM-dd HH:mm:ss SSS&quot;);
                Date date = new Date(event.getTimeStamp());
                // &#x521B;&#x5EFA;&#x65E5;&#x5FD7;&#x5E76;&#x6307;&#x5B9A;ROW KEY
                StringBuffer key = new StringBuffer(&quot;&quot;);
                if (this.sysName != null)
                    key.append(this.sysName).append(&quot;#&quot;);
                key.append(event.getLoggerName()).append(&quot;#&quot;);
                key.append(event.getLevel()).append(&quot;#&quot;);
                key.append(sdf.format(date)).append(&quot;#&quot;);
                key.append(UUID.randomUUID().toString().replace(&quot;-&quot;, &quot;&quot;));
                Put log = new Put(key.toString().getBytes());
                if (this.sysName != null)
                    log.add(hbLogFamily.getBytes(), &quot;sysName&quot;.getBytes(), this.sysName.getBytes());
                if (this.serverIp != null)
                    log.add(hbLogFamily.getBytes(), &quot;serverIp&quot;.getBytes(), this.serverIp.getBytes());
                log.add(hbLogFamily.getBytes(), &quot;LoggerName&quot;.getBytes(), event.getLoggerName().getBytes());
                log.add(hbLogFamily.getBytes(), &quot;level&quot;.getBytes(), event.getLevel().toString().getBytes());
                log.add(hbLogFamily.getBytes(), &quot;datetime&quot;.getBytes(), sdf.format(date).getBytes());
                log.add(hbLogFamily.getBytes(), &quot;message&quot;.getBytes(), event.getMessage().toString().getBytes());
                logs.add(log);
            } catch (Exception e) {
                LOGGER.error(&quot;Error logging put &quot;, e);
            }
        }
        // &#x6279;&#x91CF;&#x5199;&#x5165;HBASE
        if (logs.size() &gt; 0)
            htable.put(logs);
    }
    /**
     * &#x65E5;&#x5FD7;&#x4E8B;&#x4EF6;
     *
     * @param loggingEvent
     */
    @Override
    protected void append(LoggingEvent loggingEvent) {
        try {
            populateEvent(loggingEvent);
            // &#x6DFB;&#x52A0;&#x5230;&#x65E5;&#x5FD7;&#x961F;&#x5217;
            loggingEvents.add(loggingEvent);
        } catch (Exception e) {
            LOGGER.error(&quot;Error populating event and adding to queue&quot;, e);
        }
    }
    /**
     * &#x4E8B;&#x4EF6;&#x6D4B;&#x8BD5;
     *
     * @param event
     */
    protected void populateEvent(LoggingEvent event) {
        event.getThreadName();
        event.getRenderedMessage();
        event.getNDC();
        event.getMDCCopy();
        event.getThrowableStrRep();
        event.getLocationInformation();
    }
    @Override
    public void close() {
        try {
            if (this.task != null)
                this.task.cancel(false);
            if (this.executor != null)
                this.executor.shutdown();
            if (this.connection != null)
                this.connection.close();
            if (this.admin != null)
                this.admin.close();
            if (this.htable != null)
                this.htable.close();
        } catch (IOException e) {
            LOGGER.error(&quot;Error close &quot;, e);
        }
    }
    @Override
    public boolean requiresLayout() {
        return true;
    }
    // &#x8BBE;&#x7F6E;&#x6BCF;&#x4E00;&#x6279;&#x65E5;&#x5FD7;&#x5904;&#x7406;&#x6570;&#x91CF;
    public void setBatchSize(int batchSize) {
        this.batchSize = batchSize;
    }
    /**
     * &#x8BBE;&#x7F6E;&#x8BA1;&#x5212;&#x4EFB;&#x52A1;&#x6267;&#x884C;&#x95F4;&#x9694;
     *
     * @param period
     */
    public void setPeriod(int period) {
        this.period = period;
    }
    /**
     * &#x8BBE;&#x7F6E;&#x65E5;&#x5FD7;&#x5B58;&#x50A8;HBASE&#x8868;&#x540D;
     *
     * @param hbLogName
     */
    public void setHbLogName(String hbLogName) {
        this.hbLogName = hbLogName;
    }
    /**
     * &#x65E5;&#x5FD7;&#x8868;&#x7684;&#x5217;&#x65CF;&#x540D;&#x5B57;
     * 
     * @param hbLogFamily
     */
    public void setHbLogFamily(String hbLogFamily) {
        this.hbLogFamily = hbLogFamily;
    }
    /**
     * @param zookeeper the zookeeper to set
     */
    public void setZookeeper(String zookeeper) {
        this.zookeeper = zookeeper;
    }
    /**
     * @param sysName the sysName to set
     */
    public void setSysName(String sysName) {
        this.sysName = sysName;
    }
    /**
     * @param serverIp the serverIp to set
     */
    public void setServerIp(String serverIp) {
        this.serverIp = serverIp;
    }
}
