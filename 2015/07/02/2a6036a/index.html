2、数据字典服务，此类比较复杂，涉及的方法有create、delete、updatePart、retrieve、query、store：

使用struts2：

@Scope(&quot;prototype&quot;)
@Controller
@Namespace(&quot;/dictionary&quot;)
public class DicItemAction extends ExtJSSimpleAction&lt;DicItem&gt; {
    @Resource
    private DicService dicService;
    private String node;
    /**
     * &#x8FD4;&#x56DE;&#x6570;&#x636E;&#x5B57;&#x5178;&#x76EE;&#x5F55;&#x6811;
     * @return 
     */
    public String store() {
        if (node == null) {
            return null;
        }
        Dic dic=null;
        if(node.trim().startsWith(&quot;root&quot;)){
            dic = dicService.getRootDic();
        }else{
            int id=Integer.parseInt(node);
            dic = dicService.getDic(id);
        }
        if (dic != null) {
            String json = dicService.toJson(dic);
            Struts2Utils.renderJson(json);
        }
        return null;
    }

        this.node = node;
    }
}
