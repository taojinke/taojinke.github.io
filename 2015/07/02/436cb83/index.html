public class TPS_TEST {
    public static void main(String[] args) throws InterruptedException {
        final ClassPathXmlApplicationContext context = 
                new ClassPathXmlApplicationContext(
                        new String[] {&quot;file:E:/1-project_test/dubbox-master/dubbo-demo/dubbo-demo-consumer/src/main/resources/META-INF/spring/dubbo-demo-consumer.xml&quot;});
        final HelloService helloService = (HelloService)context.getBean(&quot;helloService&quot;); // get service invocation proxy
        ExecutorService executorServicePool = Executors.newFixedThreadPool(200);
        final int size = 100000;
        final CountDownLatch cdl = new CountDownLatch(size);
        long begin = System.currentTimeMillis();
        for (int i = 0; i &lt; size; i++) {
            executorServicePool.execute(new Runnable() {
                @Override
                public void run() {
                    try {
                        String hello = helloService.hello(&quot;aa&quot;); // do invoke!
                        //System.out.println( hello ); // cool, how are you~
                        cdl.countDown();
                    } catch (Exception e) {
                        e.printStackTrace();
                    }
                }
            });
        }
        //executorServicePool.shutdown();
        //executorService.awaitTermination(10, TimeUnit.MINUTES);
        cdl.await();//&#x7B49;&#x5F85;&#x6240;&#x6709;&#x4EFB;&#x52A1;&#x5904;&#x7406;&#x5B8C;
        long time = System.currentTimeMillis() - begin;
        System.out.println(&quot;&#x8017;&#x65F6;&#xFF1A;&quot; + (double) time / 1000 + &quot; s&quot;);
        System.out.println(&quot;&#x5E73;&#x5747;:&quot; + ((double) time) / size +&quot; ms&quot;);
        System.out.println(&quot;TPS:&quot; + (double) size / ((double) time / 1000));              
    }
}
