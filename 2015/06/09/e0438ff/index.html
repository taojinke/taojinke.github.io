using System;
using System.IO;
using System.Web.UI.WebControls;
using System.Text;
using System.Net.Mail;
using System.Net;
using System.Linq;
using System.Text.RegularExpressions;
namespace SyntacticSugar.Work
{
    /// &lt;summary&gt;     
    /// 1. &#x529F;&#x80FD;&#xFF1A;&#x53D1;&#x9001;&#x90AE;&#x7BB1;
    /// 2. &#x4F5C;&#x8005;&#xFF1A;kaixuan
    /// 3. &#x521B;&#x5EFA;&#x65E5;&#x671F;&#xFF1A;2014-10-31
    /// 4. &#x6700;&#x540E;&#x4FEE;&#x6539;&#x65E5;&#x671F;&#xFF1A;2014-10-31
    /// &lt;/summary&gt;  
    public class MailSmtp
    {
        /// &lt;summary&gt;
        /// &#x8BBE;&#x7F6E;&#x90AE;&#x4EF6;&#x7F16;&#x7801;&#x7C7B;&#x578B;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;contentEncoding&quot;&gt;&lt;/param&gt;
        public void SetEncoding(Encoding contentEncoding)
        {
            this._encoding = contentEncoding;
        }
        /// &lt;summary&gt;
        ///&#x8BBE;&#x7F6E;&#x90AE;&#x4EF6;&#x6B63;&#x6587;&#x662F;&#x5426;&#x4E3A; Html &#x683C;&#x5F0F; 
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;isHtml&quot;&gt;&lt;/param&gt;
        public void SetIsHtml(bool isHtml)
        {
            this._isHtml = isHtml;
        }
        /// &lt;summary&gt;
        /// &#x6284;&#x9001;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;cc&quot;&gt;&lt;/param&gt;
        public void SetCC(params string[] cc)
        {
            this._cc = cc;
        }
        /// &lt;summary&gt;
        /// &#x6697;&#x9001;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;cc&quot;&gt;&lt;/param&gt;
        public void SetBC(params string[] bc)
        {
            this._bcc = bc;
        }
        /// &lt;summary&gt;
        /// &#x6784;&#x9020;&#x51FD;&#x6570;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;host&quot;&gt;&lt;/param&gt;
        /// &lt;param name=&quot;username&quot;&gt;&#x90AE;&#x4EF6;&#x8D26;&#x53F7;&lt;/param&gt;
        /// &lt;param name=&quot;password&quot;&gt;&#x5BC6;&#x7801;&lt;/param&gt;
        public MailSmtp(string host, string username, string password)
        {
            this._smtp.Host = host;
            this._smtp.Port = 0x19;
            this._isHtml = true;
            this._encoding = Encoding.UTF8;
            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password))
            {
                this._smtp.UseDefaultCredentials = false;
            }
            else
            {
                this._smtp.Credentials = new NetworkCredential(username, password);
            }
        }
        /// &lt;summary&gt;
        /// &#x53D1;&#x9001;&#x90AE;&#x4EF6;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;from&quot;&gt;&#x53D1;&#x4EF6;&#x4EBA;&#x90AE;&#x4EF6;&#x5730;&#x5740;&lt;/param&gt;
        /// &lt;param name=&quot;sender&quot;&gt;&#x53D1;&#x4EF6;&#x4EBA;&#x663E;&#x793A;&#x540D;&#x79F0;&lt;/param&gt;
        /// &lt;param name=&quot;to&quot;&gt;&#x6536;&#x4EF6;&#x4EBA;&#x5730;&#x5740;&lt;/param&gt;
        /// &lt;param name=&quot;subject&quot;&gt;&#x90AE;&#x4EF6;&#x6807;&#x9898;&lt;/param&gt;
        /// &lt;param name=&quot;body&quot;&gt;&#x90AE;&#x4EF6;&#x6B63;&#x6587;&lt;/param&gt;
        /// &lt;param name=&quot;file&quot;&gt;&#x9644;&#x4EF6;&#x5730;&#x5740;&#x6570;&#x7EC4;&lt;/param&gt;
        /// &lt;returns&gt;bool &#x662F;&#x5426;&#x6210;&#x529F; &lt;/returns&gt;
        public bool Send(string from, string sender, string to, string subject, string body, params string[] file)
        {
            return Send(from, sender, new string[] { to }, subject, body, file);
        }
        /// &lt;summary&gt;
        /// &#x53D1;&#x9001;&#x90AE;&#x4EF6;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;from&quot;&gt;&#x53D1;&#x4EF6;&#x4EBA;&#x90AE;&#x4EF6;&#x5730;&#x5740;&lt;/param&gt;
        /// &lt;param name=&quot;sender&quot;&gt;&#x53D1;&#x4EF6;&#x4EBA;&#x663E;&#x793A;&#x540D;&#x79F0;&lt;/param&gt;
        /// &lt;param name=&quot;to&quot;&gt;&#x6536;&#x4EF6;&#x4EBA;&#x5730;&#x5740;&lt;/param&gt;
        /// &lt;param name=&quot;subject&quot;&gt;&#x90AE;&#x4EF6;&#x6807;&#x9898;&lt;/param&gt;
        /// &lt;param name=&quot;body&quot;&gt;&#x90AE;&#x4EF6;&#x6B63;&#x6587;&lt;/param&gt;
        /// &lt;param name=&quot;file&quot;&gt;&#x9644;&#x4EF6;&#x5730;&#x5740;&#x6570;&#x7EC4;&lt;/param&gt;
        /// &lt;returns&gt;bool &#x662F;&#x5426;&#x6210;&#x529F; &lt;/returns&gt;
        public bool Send(string from, string sender, string[] to, string subject, string body, params string[] file)
        {
            string mailReg = @&quot;^[\w-]+(\.[\w-]+)*@[\w-]+(\.[\w-]+)+$&quot;;
            if (to == null)
            {
                throw new ArgumentNullException(&quot;MailSmtp.Send.to&quot;);
            }
            if (to.Any(oit =&gt; !Regex.IsMatch(oit + &quot;&quot;, mailReg)))
            {
                this.Result = &quot;&#x6536;&#x4EF6;&#x4EBA;&#x5730;&#x5740;&#x4E0D;&#x5408;&#x6CD5;&quot;;
                return false;
            }
            if (_bcc != null &amp;&amp; _bcc.Length &gt; 0)
            {
                if (_bcc.Any(oit =&gt; !Regex.IsMatch(oit + &quot;&quot;, mailReg)))
                {
                    this.Result = &quot;&#x6697;&#x9001;&#x4EBA;&#x5730;&#x5740;&#x4E0D;&#x5408;&#x6CD5;&quot;;
                    return false;
                }
            }
            if (_cc != null &amp;&amp; _cc.Length &gt; 0)
            {
                if (_cc.Any(oit =&gt; !Regex.IsMatch(oit + &quot;&quot;, mailReg)))
                {
                    this.Result = &quot;&#x6284;&#x9001;&#x4EBA;&#x5730;&#x5740;&#x4E0D;&#x5408;&#x6CD5;&quot;;
                    return false;
                }
            }
            MailMessage message = new MailMessage();
            // &#x521B;&#x5EFA;&#x4E00;&#x4E2A;&#x9644;&#x4EF6;&#x5BF9;&#x8C61;
            foreach (var r in file)
            {
                Attachment objMailAttachment;
                objMailAttachment = new Attachment(r);//&#x53D1;&#x9001;&#x90AE;&#x4EF6;&#x7684;&#x9644;&#x4EF6;
                message.Attachments.Add(objMailAttachment);
            }
            message.From = new MailAddress(from, sender);
            message.Subject = subject;
            message.SubjectEncoding = this._encoding;
            message.Body = body;
            message.BodyEncoding = this._encoding;
            message.IsBodyHtml = this._isHtml;
            message.Priority = MailPriority.Normal;
            foreach (string str in to)
            {
                message.To.Add(str);
            }
            if (this._bcc != null &amp;&amp; this._bcc.Length &gt; 0)
            {
                foreach (string b in this._bcc)
                {
                    message.Bcc.Add(b);
                }
            }
            if (this._cc != null &amp;&amp; this._cc.Length &gt; 0)
            {
                foreach (string c in this._cc)
                {
                    message.CC.Add(c);
                }
            }
            try
            {
                this._smtp.Send(message);
                return true;
            }
            catch (Exception ex)
            {
                Console.WriteLine(ex.Message);
            }
            return false;
        }
        private SmtpClient _smtp = new SmtpClient();
        private Encoding _encoding { get; set; }
        private bool _isHtml { get; set; }
        private string[] _cc { get; set; }
        private string[] _bcc { get; set; }
        /// &lt;summary&gt;
        /// &#x83B7;&#x53D6;&#x53D1;&#x9001;&#x7ED3;&#x679C;&#xFF0C;&#x6210;&#x529F;&#x5219;&#x4E3A;&#x7A7A;
        /// &lt;/summary&gt;
        public string Result { get; private set; }
    }
}
