常见面试题系列1——原地消除字符串连续的空白字符

时间 2015-06-06 16:42:00 博客园-原创精华区

原文  http://www.cnblogs.com/webary/p/4556868.html

主题 技术

前两天看到一个文章 “ 白板编程浅谈——Why, What, How ” 里说“原地消除字符串的重复空白（例："ab   c  d e" => "ab c d e"）则是一道合格的题目，因为即便不使用库函数，合格的面试者也能够在 20 分钟内完成这道题目。” 就试着自己写了下这个功能的实现。

思路：因为题目限制了是原地，而不能借助一个辅助临时数组。所以第一想法可能会想到，从前往后遍历，遇到连续空白就将后面的内容全部向前移动，这样的确可以很快的写出来，但是时间复杂度就比较大了（0(n^2)）。所以就想到用两个游动指针来标记已经遍历到的当前位置pRight和删除空白后当前位置pLeft。

一、首先呢，我考虑写一个删除字符串中所有空格（当然特定其他字符也是一样的），就写了下面的代码：

 2 char * delChar(char* str, char del = &apos; &apos;)
 3 {
 4     //&#x9632;&#x6B62;&#x4F20;&#x5165;&#x65E0;&#x6548;&#x53C2;&#x6570;-&#x7A7A;&#x6307;&#x9488;
 5     if(str==0 || *str==0 || del==0)
 6         return str;
 7     char *pLeft=str, *pRight=str;
 8     while(*pRight!=0 &amp;&amp; *pRight!=del)
 9         ++pRight;
10     pLeft = pRight;//&#x7B2C;&#x4E00;&#x4E2A;&#x8981;&#x5220;&#x7684;&#x4F4D;&#x7F6E;
11     while(*pRight==del) //&#x8DF3;&#x8FC7;&#x8981;&#x5220;&#x9664;&#x7684;&#x5143;&#x7D20;
12         ++pRight;
13     //&#x5F53;&#x524D;pRight&#x6307;&#x5411;&#x7B2C;&#x4E00;&#x4E2A;&#x8981;&#x5411;&#x524D;&#x79FB;&#x52A8;&#x7684;&#x5143;&#x7D20;
14     for(;*pRight!=0;++pRight) {
15         //*pRight&#x4E0D;&#x9700;&#x8981;&#x5220;&#x9664;&#x5219;&#x5411;&#x524D;&#x590D;&#x5236;&#x5143;&#x7D20;
16         if(*pRight!=del) {
17             *pLeft = *pRight;
18             ++pLeft;
19         }
20     }
21     *pLeft = 0;
22     return str;
23 }

二、然后，删除连续特定字符总体来说和上面代码的框架差不多，也是用两个游动指针来指向已经遍历的位置pRight（跑的比较快，所以在右边）和删除连续字符后的当前位置pLeft（跑的比较慢），只不过是要多加一层判断，连续两个要删除的字符出现时才让pRight跑得比pLeft快。代码如下：

 1 //&#x5220;&#x9664;&#x5B57;&#x7B26;&#x6570;&#x7EC4;str&#x4E2D;&#x7684;&#x6240;&#x6709;&#x8FDE;&#x7EED;&#x91CD;&#x590D;&#x7684;&#x67D0;&#x4E2A;&#x7279;&#x5B9A;&#x5B57;&#x7B26;del
 2 char * delRepeatedChar(char* str, char del = &apos; &apos;)
 3 {
 4     //&#x9632;&#x6B62;&#x4F20;&#x5165;&#x65E0;&#x6548;&#x53C2;&#x6570;-&#x7A7A;&#x6307;&#x9488;
 5     if(str==0 || *str==0 || del==0)
 6         return str;
 7     char *pLeft=str, *pRight=str;
 8     while(*pRight!=0 &amp;&amp; *pRight!=del)
 9         ++pRight;
10     pLeft = pRight;//&#x7B2C;&#x4E00;&#x4E2A; &#x6240;&#x6307;&#x5143;&#x7D20;&#x662F;&#x8981;&#x5220;&#x9664;&#x7684;&#x5B57;&#x7B26;
11     for(; *pRight!=0; ++pLeft, ++pRight) {
12         //pRight&#x6240;&#x6307;&#x5143;&#x7D20;&#x662F;&#x8981;&#x5220;&#x9664;&#x7684;&#x5B57;&#x7B26;
13         if(*pRight==del) {
14             //&#x4E0B;&#x4E00;&#x4E2A;&#x5143;&#x7D20;&#x4E5F;&#x662F;&#x8981;&#x5220;&#x9664;&#x7684;&#x5B57;&#x7B26;&#xFF0C;&#x5219;&#x5220;&#x9664;&#x4E0B;&#x4E00;&#x4E2A;&#x5B57;&#x7B26;
15             if(*(pRight+1)==del){
16                 *pLeft++ = *pRight;
17                 //&#x79FB;&#x52A8;&#x5230;&#x5F53;&#x524D;&#x9700;&#x8981;&#x5411;&#x524D;&#x590D;&#x5236;&#x7684;&#x4F4D;&#x7F6E;
18                 while(*pRight==del)
19                     ++pRight;
20                 //&#x590D;&#x5236;&#x5143;&#x7D20;
21                 *pLeft = *pRight;
22                 continue;
23             }
24         }
25         //&#x590D;&#x5236;&#x5143;&#x7D20;
26         *pLeft = *pRight;
27     }
28     //&#x6700;&#x540E;&#x4E00;&#x4F4D;&#x7F6E;0&#xFF0C;&#x5426;&#x5219;&#x5C3E;&#x90E8;&#x53EF;&#x80FD;&#x6709;&#x4E0A;&#x4E00;&#x6B21;&#x7684;&#x6B8B;&#x7559;&#x5B57;&#x7B26;
29     *pLeft = 0;
30     return str;
31 }
