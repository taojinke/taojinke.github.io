主题 .Net

随机数在编程开发的时候使用还是比较多的，追常见的当然是验证码，除此之外，在抽奖、游戏之中也有运用。最近闲得蛋疼，就搞了个随机数生成的模板，搏总看官一乐。

先上项目目录���

简单使用：

数字不要4

RandomString rs = new RandomString();
//&#x6570;&#x5B57;&#xFF0C;&#x5E76;&#x4E14;&#x4E0D;&#x8981;4
var sdd = rs.AddProviders(EStringProviders.SingleDigit);
sdd[EStringProviders.SingleDigit][0].AddExclude(4);
do
{
    Console.WriteLine(&quot;-----------------&#x5F00;&#x59CB;---------------------&quot;);
    Stopwatch timer = new Stopwatch();
    var max = rs.TotalMember;
    timer.Start();
    string[] rev = rs.NextMany(max);
    timer.Stop();
    Console.WriteLine(string.Join(&quot;\t&quot;, rev));
    Console.WriteLine(&quot;-----------------&#x7ED3;&#x675F;---------------------&quot;);
    Console.WriteLine(&quot;&#x8017;&#x65F6;{0}, &#x4EA7;&#x751F;{1}&#x4E2A;&#x968F;&#x673A;&#x5BF9;&#x8C61;&quot;, timer.Elapsed.TotalMilliseconds, rev.Length); 
}while (Console.ReadKey().Key != ConsoleKey.Escape);

字母大小写

RandomString rs = new RandomString();
rs.AddProviders(EStringProviders.Lowercase | EStringProviders.Uppercase);
do
{
    Console.WriteLine(&quot;-----------------&#x5F00;&#x59CB;---------------------&quot;);
    Stopwatch timer = new Stopwatch();
    var max = rs.TotalMember;
    timer.Start();
    string[] rev = rs.NextMany(max);
    timer.Stop();
    Console.WriteLine(string.Join(&quot;\t&quot;, rev));
    Console.WriteLine(&quot;-----------------&#x7ED3;&#x675F;---------------------&quot;);
    Console.WriteLine(&quot;&#x8017;&#x65F6;{0}, &#x4EA7;&#x751F;{1}&#x4E2A;&#x968F;&#x673A;&#x5BF9;&#x8C61;&quot;, timer.Elapsed.TotalMilliseconds, rev.Length); 
}
while (Console.ReadKey().Key != ConsoleKey.Escape);
