实现

using System;
using System.Collections.Generic;
using System.Drawing;
using System.IO;
using System.Linq;
using System.Runtime.Serialization;
using System.ServiceModel;
using System.ServiceModel.Web;
using System.Text;
namespace Wolfy.WCFRestfuleDemo
{
    // NOTE: You can use the &quot;Rename&quot; command on the &quot;Refactor&quot; menu to change the class name &quot;ImageService&quot; in code, svc and config file together.
    // NOTE: In order to launch WCF Test Client for testing this service, please select ImageService.svc or ImageService.svc.cs at the Solution Explorer and start debugging.
    public class ImageService : IImageService
    {
        /// &lt;summary&gt;
        /// &#x6839;&#x636E;&#x56FE;&#x7247;&#x7684;&#x76F8;&#x5BF9;&#x8DEF;&#x5F84;&#x83B7;&#x53D6;&#x6587;&#x4EF6;&#x6D41;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;imgUrl&quot;&gt;&lt;/param&gt;
        /// &lt;returns&gt;&lt;/returns&gt;
        public System.IO.Stream GetImageStream(string imgUrl)
        {
            var contentType = Path.GetExtension(imgUrl).Trim(&apos;.&apos;);
            WebOperationContext woc = WebOperationContext.Current;
            //&#x6839;&#x636E;&#x8BF7;&#x6C42;&#x7684;&#x56FE;&#x7247;&#x7C7B;&#x578B;&#xFF0C;&#x52A8;&#x6001;&#x8BBE;&#x7F6E;contenttype
            woc.OutgoingResponse.ContentType = &quot;image/&quot; + contentType;
            string savePath = System.Web.HttpContext.Current.Server.MapPath(&quot;/Images&quot;);
            string filePath = Path.Combine(savePath, imgUrl);
            return File.OpenRead(filePath);
        }
        /// &lt;summary&gt;
        /// &#x4E0A;&#x4F20;&#x56FE;&#x7247;
        /// &lt;/summary&gt;
        /// &lt;param name=&quot;imgStream&quot;&gt;&lt;/param&gt;
        /// &lt;param name=&quot;imageName&quot;&gt;&lt;/param&gt;
        public void UploadImage(System.IO.Stream imgStream, string imageName)
        {
            var dir = System.Web.HttpContext.Current.Server.MapPath(&quot;~/Images&quot;);
            var file = Path.Combine(dir, imageName);
            var bitmap = Bitmap.FromStream(imgStream);
            bitmap.Save(file);
        }
        /// &lt;summary&gt;
        /// &#x83B7;&#x5F97;&#x6240;&#x6709;&#x56FE;&#x7247;&#x7684;&#x76F8;&#x5BF9;&#x8DEF;&#x5F84;
        /// &lt;/summary&gt;
        /// &lt;returns&gt;&lt;/returns&gt;
        public string[] GetImages()
        {
            List&lt;string&gt; lstImages = new List&lt;string&gt;();
            var dir = System.Web.HttpContext.Current.Server.MapPath(&quot;~/Images&quot;);
            string[] paths = Directory.GetFiles(dir);
            for (int i = 0; i &lt; paths.Length; i++)
            {
                lstImages.Add(paths[i].Replace(dir, &quot;&quot;));
            }
            return lstImages.ToArray();
        }
    }
}

客户端提示

查看Images目录，1.jpg已经上传成功。

通过restful服务在浏览器中查看：在浏览器中发送get请求，将会调用GetImageStream方法，将stream响应给浏览器，浏览器进行渲染。

还剩最后一个接口测试，返回所有的图片。因为wcf寄宿的也是一个web站点，所以也可以通过在浏览器中直接调用，将会返回所有的图片的相对路径的xml信息并在页面上进行展示。

总结

本文介绍了restful接口如何处理post过来的stream，以及如何返回stream给客户端的方式，这里也是一种上传下载文件的一种方式。
